class Node(object):
    def __init__(self, key, end = None):
        self.key = key
        self.end = end
        self.child = {}

class Trie:
    def __init__(self):
        self.head = Node(self)
    
    def insert(self, word):
        cur = self.head

        for w in word:
            if w not in cur.child:
                cur.child[w] = Node(w)
            cur = cur.child[w]
        cur.end = True

    def search(self, word):
        cur = self.head

        for w in word:
            if w in cur.child:
                cur = cur.child[w]
            else:
                return False

        if cur.end == True:
            return True
